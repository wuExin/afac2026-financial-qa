# BM25 Debug UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local BM25 debugging page where a user can upload or paste question JSON, tune retrieval parameters, run search, and inspect retrieved chunks.

**Architecture:** Keep retrieval logic in a testable Python service module, expose it through a small standard-library HTTP server, and render the tool with a static HTML/CSS/JavaScript page. The service reuses `FinancialQAAgent._load_evidence()` and `BM25Retriever`, disables retrieval logging for UI searches, and returns JSON shaped for direct rendering.

**Tech Stack:** Python standard library `http.server`, existing `src.agent.agent` classes, existing YAML config loader, vanilla HTML/CSS/JavaScript, `pytest`.

## Global Constraints

- Do not add third-party dependencies.
- The tool is local-only and does not call the LLM.
- The page supports both a single question object and an array from `*_questions.json`; the backend accepts exactly one selected question per search request.
- The tunable parameters are `bm25_k1`, `bm25_b`, `chunk_size_chars`, `chunk_overlap_chars`, `min_chunk_chars`, `expand_before_chars`, `expand_after_chars`, `merge_gap_chars`, `per_doc_min`, `per_doc_max`, `global_top_k`, `max_total_chars`, `min_score`, and `max_query_terms`.
- The UI uses single numeric overrides for `chunk_size_chars`, `per_doc_max`, `global_top_k`, and `max_total_chars`.
- `retrieval_logger` is disabled for debug searches so repeated page use does not overwrite `logs/<qid>.json`.
- Default server address is `http://127.0.0.1:8765`.

---

## File Structure

- Create `src/agent/bm25_debug_service.py`
  - Owns request validation, parameter coercion, config merging, retrieval execution, and response serialization.
  - Has no HTTP or browser responsibilities.
- Create `scripts/bm25_debug_server.py`
  - Owns CLI args, local HTTP server lifecycle, static page serving, and JSON API wiring.
  - Delegates all retrieval behavior to `src.agent.bm25_debug_service`.
- Create `scripts/bm25_debug_ui.html`
  - Owns upload/paste UI, question selection, parameter controls, `/api/search` calls, and result rendering.
- Create `tests/test_bm25_debug_service.py`
  - Tests backend validation, parameter overrides, successful retrieval, and invalid input responses without opening a network socket.
- Create `tests/test_bm25_debug_server.py`
  - Tests HTTP handler helpers for JSON success/error responses and static file serving through direct handler methods where practical.

```python
"""Service helpers for the local BM25 debug UI."""

from __future__ import annotations

from copy import deepcopy
from typing import Any, Dict, List, Optional

from src.agent.agent import BM25Retriever, Evidence, FinancialQAAgent
from src.utils.helpers import load_config


FLOAT_PARAM_KEYS = {"bm25_k1", "bm25_b", "min_score"}
INT_PARAM_KEYS = {
    "chunk_size_chars",
    "chunk_overlap_chars",
    "min_chunk_chars",
    "expand_before_chars",
    "expand_after_chars",
    "merge_gap_chars",
    "per_doc_min",
    "per_doc_max",
    "global_top_k",
    "max_total_chars",
    "max_query_terms",
}
ALLOWED_PARAM_KEYS = FLOAT_PARAM_KEYS | INT_PARAM_KEYS


def validate_question(question: object) -> Dict[str, Any]:
    if not isinstance(question, dict):
        raise ValueError("question 必须是对象")
    if not question.get("domain"):
        raise ValueError("question.domain 不能为空")
    if not question.get("question"):
        raise ValueError("question.question 不能为空")
    doc_ids = question.get("doc_ids")
    if not isinstance(doc_ids, list) or not doc_ids:
        raise ValueError("question.doc_ids 不能为空")
    return question


def build_retrieval_config(
    base_config: Dict[str, Any],
    params: Optional[Dict[str, Any]],
) -> Dict[str, Any]:
    retrieval_cfg = deepcopy(base_config.get("retrieval", {}))
    params = params or {}
    for key, value in params.items():
        if key in FLOAT_PARAM_KEYS:
            retrieval_cfg[key] = float(value)
        elif key in INT_PARAM_KEYS:
            retrieval_cfg[key] = int(value)
    retrieval_cfg["logging"] = {"log_retrieval": False}
    return retrieval_cfg


def serialize_evidence(evidence: Evidence) -> Dict[str, Any]:
    return {
        "doc_id": evidence.doc_id,
        "source": evidence.source,
        "relevance_score": evidence.relevance_score,
        "content": evidence.content,
    }


def serialize_debug_response(
    question: Dict[str, Any],
    retrieved: List[Evidence],
    stats: Dict[str, Any],
) -> Dict[str, Any]:
    chunks = []
    selected_sources = stats.get("selected_sources", [])
    source_by_key = {
        (str(item.get("doc_id")), int(item.get("start", 0)), int(item.get("end", 0))): item
        for item in selected_sources
    }
    for evidence in retrieved:
        for key, source in source_by_key.items():
            if key[0] != str(evidence.doc_id):
                continue
            marker = f"位置 {key[1]}-{key[2]}"
            if marker not in evidence.content:
                continue
            chunks.append(
                {
                    "doc_id": evidence.doc_id,
                    "source": evidence.source,
                    "start": source.get("start", 0),
                    "end": source.get("end", 0),
                    "score": source.get("score", 0.0),
                    "query_types": source.get("query_types", []),
                    "text": evidence.content,
                }
            )
            break
    return {
        "ok": True,
        "question": {
            "qid": question.get("qid", ""),
            "domain": question.get("domain", ""),
            "answer_format": question.get("answer_format", ""),
            "doc_ids": question.get("doc_ids", []),
            "question": question.get("question", ""),
            "options": question.get("options", {}),
        },
        "stats": stats,
        "chunks": chunks,
        "retrieved": [serialize_evidence(item) for item in retrieved],
    }


def run_debug_search(
    question: Dict[str, Any],
    params: Optional[Dict[str, Any]] = None,
    config: Optional[Dict[str, Any]] = None,
) -> Dict[str, Any]:
    question = validate_question(question)
    base_config = deepcopy(config or load_config())
    retrieval_cfg = build_retrieval_config(base_config, params)
    agent = FinancialQAAgent(base_config)
    evidence = agent._load_evidence(question)
    retriever = BM25Retriever(retrieval_cfg)
    retrieved, stats = retriever.retrieve(question, evidence)
    return serialize_debug_response(question, retrieved, stats)
```

"""Local BM25 debug UI server."""

from __future__ import annotations

import argparse
import json
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer
from pathlib import Path
from typing import Any, Dict

from src.agent.bm25_debug_service import run_debug_search

ROOT = Path(__file__).resolve().parents[1]
UI_PATH = ROOT / "scripts" / "bm25_debug_ui.html"


def parse_json_body(raw: bytes) -> Dict[str, Any]:
    try:
        payload = json.loads(raw.decode("utf-8"))
    except json.JSONDecodeError as exc:
        raise ValueError(f"JSON 解析失败: {exc.msg}") from exc
    if not isinstance(payload, dict):
        raise ValueError("请求体必须是 JSON 对象")
    return payload


def make_error(message: str) -> Dict[str, Any]:
    return {"ok": False, "error": message}


class BM25DebugHandler(BaseHTTPRequestHandler):
    server_version = "BM25Debug/1.0"

    def do_GET(self) -> None:
        if self.path in ("/", "/index.html"):
            self._send_file(UI_PATH, "text/html; charset=utf-8")
            return
        self._send_json(make_error("Not found"), status=404)

    def do_POST(self) -> None:
        if self.path != "/api/search":
            self._send_json(make_error("Not found"), status=404)
            return
        try:
            length = int(self.headers.get("Content-Length", "0"))
            payload = parse_json_body(self.rfile.read(length))
            question = payload.get("question")
            params = payload.get("params", {})
            result = run_debug_search(question, params)
            self._send_json(result)
        except Exception as exc:
            self._send_json(make_error(str(exc)), status=400)

    def log_message(self, fmt: str, *args: object) -> None:
        print(f"[BM25Debug] {self.address_string()} - {fmt % args}")

    def _send_file(self, path: Path, content_type: str) -> None:
        if not path.is_file():
            self._send_json(make_error(f"静态文件不存在: {path}"), status=404)
            return
        content = path.read_bytes()
        self.send_response(200)
        self.send_header("Content-Type", content_type)
        self.send_header("Content-Length", str(len(content)))
        self.end_headers()
        self.wfile.write(content)

    def _send_json(self, payload: Dict[str, Any], status: int = 200) -> None:
        content = json.dumps(payload, ensure_ascii=False, indent=2).encode("utf-8")
        self.send_response(status)
        self.send_header("Content-Type", "application/json; charset=utf-8")
        self.send_header("Content-Length", str(len(content)))
        self.end_headers()
        self.wfile.write(content)


def main() -> None:
    parser = argparse.ArgumentParser(description="BM25 本地调试页面")
    parser.add_argument("--host", default="127.0.0.1")
    parser.add_argument("--port", type=int, default=8765)
    args = parser.parse_args()

    server = ThreadingHTTPServer((args.host, args.port), BM25DebugHandler)
    url = f"http://{args.host}:{args.port}"
    print(f"BM25 debug UI: {url}")
    try:
        server.serve_forever()
    except KeyboardInterrupt:
        print("\nBM25 debug server stopped")
    finally:
        server.server_close()


if __name__ == "__main__":
    main()
```

      min_score: 0.1,
      max_query_terms: 120
    };
    let questions = [];
    let selectedIndex = -1;

    const el = (id) => document.getElementById(id);
    const escapeHtml = (value) => String(value ?? "").replace(/[&<>"']/g, ch => ({
      "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#039;"
    }[ch]));

    function showError(message) {
      const box = el("errorBox");
      box.textContent = message;
      box.style.display = message ? "block" : "none";
    }

    function normalizeQuestions(value) {
      if (Array.isArray(value)) return value;
      if (value && typeof value === "object") return [value];
      throw new Error("JSON 必须是单题对象或题目数组");
    }

    function parseJsonInput() {
      const parsed = JSON.parse(el("jsonInput").value);
      questions = normalizeQuestions(parsed);
      selectedIndex = questions.length ? 0 : -1;
      renderQuestionList();
      showError("");
    }

    function renderQuestionList() {
      const filter = el("filterInput").value.trim().toLowerCase();
      const list = el("questionList");
      list.innerHTML = "";
      questions.forEach((q, index) => {
        const haystack = `${q.qid || ""} ${q.domain || ""} ${q.answer_format || ""}`.toLowerCase();
        if (filter && !haystack.includes(filter)) return;
        const btn = document.createElement("button");
        btn.type = "button";
        btn.className = `question-item${index === selectedIndex ? " active" : ""}`;
        btn.innerHTML = `<strong>${escapeHtml(q.qid || `#${index + 1}`)}</strong><br>${escapeHtml(q.domain || "")} · ${escapeHtml(q.answer_format || "")}`;
        btn.onclick = () => {
          selectedIndex = index;
          renderQuestionList();
        };
        list.appendChild(btn);
      });
      if (!list.children.length) {
        list.innerHTML = '<div class="empty">没有匹配题目</div>';
      }
    }

    function renderParams() {
      const form = el("paramsForm");
      form.innerHTML = "";
      Object.entries(defaultParams).forEach(([key, value]) => {
        const wrap = document.createElement("div");
        wrap.innerHTML = `<label for="param_${key}">${key}</label><input id="param_${key}" type="number" step="any" value="${value}">`;
        form.appendChild(wrap);
      });
    }

    function collectParams() {
      const params = {};
      Object.keys(defaultParams).forEach(key => {
        const value = el(`param_${key}`).value;
        if (value !== "") params[key] = Number(value);
      });
      return params;
    }

    async function search() {
      if (selectedIndex < 0 || !questions[selectedIndex]) {
        showError("请先解析并选择一道题");
        return;
      }
      showError("");
      el("results").className = "empty";
      el("results").textContent = "搜索中";
      const response = await fetch("/api/search", {
        method: "POST",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({question: questions[selectedIndex], params: collectParams()})
      });
      const payload = await response.json();
      if (!payload.ok) {
        showError(payload.error || "搜索失败");
        el("results").textContent = "搜索失败";
        return;
      }
      renderStats(payload.stats || {});
      renderResults(payload);
    }

    function renderStats(stats) {
      const keys = ["query_count", "chunk_count", "candidate_count", "retrieved_windows", "doc_coverage", "max_bm25_score", "avg_bm25_score"];
      el("stats").innerHTML = `<div class="stat-grid">${keys.map(key => `
        <div class="stat"><strong>${escapeHtml(formatNumber(stats[key]))}</strong><span>${key}</span></div>
      `).join("")}</div>`;
    }

    function formatNumber(value) {
      return typeof value === "number" ? value.toFixed(value % 1 === 0 ? 0 : 4) : (value ?? "");
    }

    function renderResults(payload) {
      const chunks = payload.chunks && payload.chunks.length ? payload.chunks : payload.retrieved || [];
      if (!chunks.length) {
        el("results").className = "empty";
        el("results").textContent = "没有命中片段";
        return;
      }
      el("results").className = "";
      el("results").innerHTML = chunks.map(chunk => `
        <article class="chunk">
          <div class="meta">
            <span>doc_id=${escapeHtml(chunk.doc_id)}</span>
            <span>source=${escapeHtml(chunk.source)}</span>
            <span>pos=${escapeHtml(chunk.start ?? "")}-${escapeHtml(chunk.end ?? "")}</span>
            <span>score=${escapeHtml(formatNumber(chunk.score ?? chunk.relevance_score))}</span>
            <span>query=${escapeHtml((chunk.query_types || []).join(","))}</span>
          </div>
          <pre>${escapeHtml(chunk.text || chunk.content || "")}</pre>
        </article>
      `).join("");
    }

    el("parseBtn").onclick = () => {
      try { parseJsonInput(); } catch (err) { showError(err.message); }
    };
    el("filterInput").oninput = renderQuestionList;
    el("searchBtn").onclick = () => search().catch(err => showError(err.message));
    el("resetParamsBtn").onclick = renderParams;
    el("fileInput").onchange = async (event) => {
      const file = event.target.files[0];
      if (!file) return;
      el("jsonInput").value = await file.text();
      try { parseJsonInput(); } catch (err) { showError(err.message); }
    };
    renderParams();
  </script>
</body>
</html>
```
